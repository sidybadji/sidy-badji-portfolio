Vue d'ensemble du projet

Pipeline de données complet pour l'analyse prédictive des maladies cardiaques utilisant Apache Spark et Python, avec des modèles de Machine Learning pour la classification des risques.

Problématique

  • Volume de données : Traitement de 300,000+ enregistrements de données médicales
  • Complexité : 13+ caractéristiques cliniques à analyser simultanément
  • Performance : Besoin de traitement rapide pour des analyses en temps réel
  • Précision : Classification binaire avec haute précision pour l'aide au diagnostic

Solution

Pipeline de données distribuées avec Apache Spark pour :

  • Ingestion : Chargement et validation des données médicales
  • Transformation : Feature engineering et normalisation des données
  • Modélisation : Entraînement de modèles ML avec MLlib
  • Prédiction : Classification des risques cardiaques en temps réel

Extraits de code

1. Ingestion des données
from pyspark.sql import SparkSession
from pyspark.sql.types import *
from pyspark.sql.functions import *

# Initialisation de Spark
spark = SparkSession.builder \
    .appName("HeartDiseaseAnalysis") \
    .config("spark.sql.adaptive.enabled", "true") \
    .config("spark.sql.adaptive.coalescePartitions.enabled", "true") \
    .getOrCreate()

# Définition du schéma
schema = StructType([
    StructField("age", IntegerType(), True),
    StructField("sex", IntegerType(), True),
    StructField("cp", IntegerType(), True),
    StructField("trestbps", IntegerType(), True),
    StructField("chol", IntegerType(), True),
    StructField("fbs", IntegerType(), True),
    StructField("restecg", IntegerType(), True),
    StructField("thalach", IntegerType(), True),
    StructField("exang", IntegerType(), True),
    StructField("oldpeak", DoubleType(), True),
    StructField("slope", IntegerType(), True),
    StructField("ca", IntegerType(), True),
    StructField("thal", IntegerType(), True),
    StructField("target", IntegerType(), True)
])

# Chargement des données
df = spark.read \
    .schema(schema) \
    .option("header", "true") \
    .csv("data/heart_disease.csv")

print(f"Nombre d'enregistrements: {df.count()}")
df.printSchema()
2. Nettoyage des données
from pyspark.sql.functions import when, col, isnan, count

# Vérification des valeurs manquantes
missing_data = df.select([count(when(col(c).isNull() | isnan(col(c)), c)).alias(c) 
                         for c in df.columns])
missing_data.show()

# Gestion des valeurs aberrantes
df_cleaned = df.filter(
    (col("age") >= 29) & (col("age") <= 77) &
    (col("trestbps") >= 94) & (col("trestbps") <= 200) &
    (col("chol") >= 126) & (col("chol") <= 564) &
    (col("thalach") >= 71) & (col("thalach") <= 202) &
    (col("oldpeak") >= 0) & (col("oldpeak") <= 6.2)
)

# Suppression des doublons
df_cleaned = df_cleaned.dropDuplicates()

print(f"Données après nettoyage: {df_cleaned.count()}")
3. Feature Engineering
from pyspark.ml.feature import VectorAssembler, StandardScaler, StringIndexer
from pyspark.ml import Pipeline

# Création de nouvelles features
df_features = df_cleaned.withColumn(
    "age_group", 
    when(col("age") < 45, 0)
    .when(col("age") < 55, 1)
    .when(col("age") < 65, 2)
    .otherwise(3)
).withColumn(
    "bmi_category",
    when(col("chol") / (col("age") * 0.01) < 25, 0)
    .when(col("chol") / (col("age") * 0.01) < 30, 1)
    .otherwise(2)
).withColumn(
    "risk_factor",
    (col("age") * 0.1) + (col("chol") * 0.01) + 
    (col("trestbps") * 0.1) + (col("oldpeak") * 10)
)

# Préparation des features pour ML
feature_columns = ["age", "sex", "cp", "trestbps", "chol", "fbs", 
                   "restecg", "thalach", "exang", "oldpeak", "slope", 
                   "ca", "thal", "age_group", "bmi_category", "risk_factor"]

# Assemblage des vecteurs
assembler = VectorAssembler(
    inputCols=feature_columns,
    outputCol="features"
)

# Normalisation
scaler = StandardScaler(
    inputCol="features",
    outputCol="scaled_features"
)

# Pipeline de transformation
transformation_pipeline = Pipeline(stages=[assembler, scaler])
df_transformed = transformation_pipeline.fit(df_features).transform(df_features)
4. Entraînement des modèles ML
from pyspark.ml.classification import RandomForestClassifier, LogisticRegression, GBTClassifier
from pyspark.ml.evaluation import BinaryClassificationEvaluator, MulticlassClassificationEvaluator
from pyspark.ml.tuning import CrossValidator, ParamGridBuilder

# Division train/test
train_data, test_data = df_transformed.randomSplit([0.8, 0.2], seed=42)

# Modèle Random Forest
rf = RandomForestClassifier(
    featuresCol="scaled_features",
    labelCol="target",
    numTrees=100,
    maxDepth=10,
    seed=42
)

# Modèle Logistic Regression
lr = LogisticRegression(
    featuresCol="scaled_features",
    labelCol="target",
    maxIter=100,
    regParam=0.01
)

# Modèle Gradient Boosting
gbt = GBTClassifier(
    featuresCol="scaled_features",
    labelCol="target",
    maxIter=50
)

# Entraînement des modèles
rf_model = rf.fit(train_data)
lr_model = lr.fit(train_data)
gbt_model = gbt.fit(train_data)

# Prédictions
rf_predictions = rf_model.transform(test_data)
lr_predictions = lr_model.transform(test_data)
gbt_predictions = gbt_model.transform(test_data)
5. Évaluation des modèles
from pyspark.ml.evaluation import BinaryClassificationEvaluator, MulticlassClassificationEvaluator

# Évaluateurs
binary_evaluator = BinaryClassificationEvaluator(labelCol="target")
multi_evaluator = MulticlassClassificationEvaluator(labelCol="target")

def evaluate_model(predictions, model_name):
    """Évalue un modèle et retourne les métriques"""
    
    # AUC
    auc = binary_evaluator.evaluate(predictions, {binary_evaluator.metricName: "areaUnderROC"})
    
    # Précision, Recall, F1
    accuracy = multi_evaluator.evaluate(predictions, {multi_evaluator.metricName: "accuracy"})
    precision = multi_evaluator.evaluate(predictions, {multi_evaluator.metricName: "weightedPrecision"})
    recall = multi_evaluator.evaluate(predictions, {multi_evaluator.metricName: "weightedRecall"})
    f1 = multi_evaluator.evaluate(predictions, {multi_evaluator.metricName: "f1"})
    
    print(f"\n=== {model_name} ===")
    print(f"AUC: {auc:.4f}")
    print(f"Accuracy: {accuracy:.4f}")
    print(f"Precision: {precision:.4f}")
    print(f"Recall: {recall:.4f}")
    print(f"F1-Score: {f1:.4f}")
    
    return {"auc": auc, "accuracy": accuracy, "precision": precision, "recall": recall, "f1": f1}

# Évaluation de tous les modèles
rf_metrics = evaluate_model(rf_predictions, "Random Forest")
lr_metrics = evaluate_model(lr_predictions, "Logistic Regression")
gbt_metrics = evaluate_model(gbt_predictions, "Gradient Boosting")

# Sélection du meilleur modèle
best_model = max([rf_metrics, lr_metrics, gbt_metrics], key=lambda x: x['f1'])
print(f"\nMeilleur modèle: {best_model}")
6. Déploiement et prédiction
import pickle
from pyspark.ml import PipelineModel

# Sauvegarde du modèle
best_model_path = "models/heart_disease_best_model"
if best_model == rf_metrics:
    rf_model.write().overwrite().save(best_model_path)
    model_type = "RandomForest"
elif best_model == lr_metrics:
    lr_model.write().overwrite().save(best_model_path)
    model_type = "LogisticRegression"
else:
    gbt_model.write().overwrite().save(best_model_path)
    model_type = "GradientBoosting"

print(f"Modèle {model_type} sauvegardé avec succès")

# Fonction de prédiction en temps réel
def predict_heart_disease(patient_data):
    """Prédit le risque de maladie cardiaque pour un patient"""
    
    # Chargement du modèle
    model = PipelineModel.load(best_model_path)
    
    # Création du DataFrame patient
    patient_df = spark.createDataFrame([patient_data], schema)
    
    # Prédiction
    prediction = model.transform(patient_df)
    
    # Récupération du résultat
    result = prediction.select("prediction", "probability").collect()[0]
    
    risk_level = "Élevé" if result["prediction"] == 1 else "Faible"
    confidence = result["probability"][1] if result["prediction"] == 1 else result["probability"][0]
    
    return {
        "risk_level": risk_level,
        "confidence": round(confidence * 100, 2),
        "recommendation": "Consultation cardiologique recommandée" if result["prediction"] == 1 else "Surveillance normale"
    }

# Exemple d'utilisation
sample_patient = {
    "age": 55, "sex": 1, "cp": 3, "trestbps": 145,
    "chol": 233, "fbs": 1, "restecg": 0, "thalach": 150,
    "exang": 0, "oldpeak": 2.3, "slope": 0, "ca": 0, "thal": 1, "target": 1
}

result = predict_heart_disease(sample_patient)
print(f"Résultat: {result}")
