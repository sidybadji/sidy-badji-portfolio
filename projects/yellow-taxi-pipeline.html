from pyspark.sql import SparkSession
from pyspark.sql.functions import *
from pyspark.sql.types import *

def process_taxi_data():
    spark = SparkSession.builder \
        .appName("TaxiDataProcessing") \
        .getOrCreate()
    
    # Lecture des donn√©es depuis Cloud Storage
    df = spark.read.option("header", "true") \
        .csv("gs://taxi-data-lake/raw-data/*.csv")
    
    # Transformation et nettoyage
    processed_df = df \
        .filter(col("trip_distance") > 0) \
        .withColumn("trip_duration", 
                   unix_timestamp(col("dropoff_datetime")) - 
                   unix_timestamp(col("pickup_datetime"))) \
        .withColumn("hour_of_day", hour(col("pickup_datetime"))) \
        .withColumn("day_of_week", dayofweek(col("pickup_datetime")))
    
    # √âcriture vers BigQuery
    processed_df.write \
        .format("bigquery") \
        .option("table", "taxi_analytics.processed_trips") \
        .mode("overwrite") \
        .save()

if __name__ == "__main__":
    process_taxi_data()

from __future__ import annotations
import logging
from datetime import datetime, UTC
import requests
from time import sleep
from google.cloud import storage
from config import (
    PROJECT_ID, BUCKET_NAME, GCS_TRIPS_PREFIX, upload_log_to_gcs, storage_client
)

BASE_URL = "https://d37ci6vzurychx.cloudfront.net/trip-data"
SESSION = requests.Session()
SESSION.headers.update({"User-Agent": "nyc-yellow-downloader/1.0"})

def stream_download_to_gcs(url: str, bucket: storage.Bucket, gcs_path: str) -> None:
    with SESSION.get(url, stream=True, timeout=60) as r:
        if r.status_code == 404:
            logging.warning(f"Source 404: {url}")
            return
        r.raise_for_status()
        blob = bucket.blob(gcs_path)
        with blob.open("wb") as f:
            for chunk in r.iter_content(chunk_size=1024 * 1024):
                if chunk:
                    f.write(chunk)

def run(start_year: int = 2020) -> None:
    logging.info(f"[DL] Start historical download @ {datetime.now(UTC).isoformat()}")
    bucket = storage_client().bucket(BUCKET_NAME)
    now = datetime.now(UTC)
    current_year = now.year
    current_month = now.month

    for year in range(start_year, current_year + 1):
        for month in range(1, 13):
            if year == current_year and month > current_month:
                break
            file_name = f"yellow_tripdata_{year}-{month:02d}.parquet"
            gcs_path = f"{GCS_TRIPS_PREFIX}{file_name}"
            url = f"{BASE_URL}/{file_name}"

            logging.info(f"[DL] {url} ‚Üí gs://{BUCKET_NAME}/{gcs_path}")
            try:
                stream_download_to_gcs(url, bucket, gcs_path)
                logging.info(f"[DL] OK {file_name}")
                sleep(1)
            except Exception as e:
                logging.error(f"[DL] Error for {file_name}: {e}")

from __future__ import annotations
import logging
from google.cloud import bigquery
from config import (
    BUCKET_NAME, GCS_TRIPS_PREFIX, RAW_TABLE, TEMP_TABLE, upload_log_to_gcs,
    storage_client, bq_client
)

def load_one_file(file_name: str) -> None:
    client = bq_client()
    uri = f"gs://{BUCKET_NAME}/{GCS_TRIPS_PREFIX}{file_name}"
    logging.info(f"[LOAD] {uri} ‚Üí {TEMP_TABLE}")

    load_cfg = bigquery.LoadJobConfig(
        source_format=bigquery.SourceFormat.PARQUET,
        write_disposition=bigquery.WriteDisposition.WRITE_TRUNCATE,
        autodetect=True,
    )
    client.load_table_from_uri(uri, TEMP_TABLE, job_config=load_cfg).result()
    
    insert_sql = f"""
    INSERT INTO `{RAW_TABLE}` (
        VendorID, tpep_pickup_datetime, tpep_dropoff_datetime, passenger_count,
        trip_distance, RatecodeID, store_and_fwd_flag, PULocationID, DOLocationID,
        payment_type, fare_amount, extra, mta_tax, tip_amount, tolls_amount,
        improvement_surcharge, total_amount, congestion_surcharge, airport_fee, source_file
    )
    SELECT 
        VendorID, tpep_pickup_datetime, tpep_dropoff_datetime,
        CAST(passenger_count AS FLOAT64), trip_distance, RatecodeID,
        store_and_fwd_flag, PULocationID, DOLocationID, payment_type,
        fare_amount, extra, mta_tax, tip_amount, tolls_amount,
        improvement_surcharge, total_amount, congestion_surcharge,
        airport_fee, @src AS source_file
    FROM `{TEMP_TABLE}`
    """
    job = client.query(insert_sql, job_config=bigquery.QueryJobConfig(
        query_parameters=[bigquery.ScalarQueryParameter("src", "STRING", file_name)]
    ))
    job.result()
    logging.info(f"[LOAD] done {file_name}")
