import pandas as pd
import numpy as np
from sklearn.preprocessing import StandardScaler, LabelEncoder
from sklearn.ensemble import RandomForestClassifier
from sklearn.model_selection import train_test_split
import warnings
warnings.filterwarnings('ignore')

class BankingRiskAssessment:
    """Syst√®me d'√©valuation des risques bancaires"""
    
    def __init__(self):
        self.scaler = StandardScaler()
        self.label_encoders = {}
        self.model = RandomForestClassifier(
            n_estimators=100,
            max_depth=15,
            min_samples_split=10,
            random_state=42
        )
    
    def load_data(self, file_path):
        """Chargement des donn√©es bancaires"""
        print("üìä Chargement des donn√©es bancaires...")
        
        # Simulation de donn√©es bancaires
        np.random.seed(42)
        n_samples = 10000
        
        data = {
            'customer_id': range(1, n_samples + 1),
            'age': np.random.normal(35, 12, n_samples).astype(int),
            'income': np.random.lognormal(10, 0.5, n_samples),
            'credit_score': np.random.normal(650, 100, n_samples).astype(int),
            'loan_amount': np.random.exponential(50000, n_samples),
            'employment_length': np.random.exponential(5, n_samples),
            'debt_to_income': np.random.beta(2, 5, n_samples) * 0.6,
            'payment_history': np.random.choice(['excellent', 'good', 'fair', 'poor'], n_samples),
            'account_balance': np.random.normal(10000, 5000, n_samples),
            'default_risk': np.random.choice([0, 1], n_samples, p=[0.85, 0.15])
        }
        
        self.df = pd.DataFrame(data)
        print(f"‚úÖ {len(self.df)} clients charg√©s")
        return self.df
    
    def feature_engineering(self):
        """Ing√©nierie des caract√©ristiques avanc√©e"""
        print("üîß Feature Engineering en cours...")
        
        df = self.df.copy()
        
        # 1. Variables num√©riques
        numerical_features = ['age', 'income', 'credit_score', 'loan_amount', 
                            'employment_length', 'debt_to_income', 'account_balance']
        
        # 2. Encodage des variables cat√©gorielles
        categorical_features = ['payment_history']
        for feature in categorical_features:
            if feature not in self.label_encoders:
                self.label_encoders[feature] = LabelEncoder()
            df[f'{feature}_encoded'] = self.label_encoders[feature].fit_transform(df[feature])
        
        # 3. Nouvelles caract√©ristiques d√©riv√©es
        df['income_to_loan_ratio'] = df['income'] / (df['loan_amount'] + 1)
        df['credit_utilization'] = df['debt_to_income'] * df['account_balance']
        df['employment_stability'] = df['employment_length'] / (df['age'] - 18 + 1)
        df['financial_health'] = (df['credit_score'] / 850) * (1 - df['debt_to_income'])
        
        # 4. Cat√©gorisation de l'√¢ge
        df['age_group'] = pd.cut(df['age'], bins=[0, 25, 35, 50, 100], 
                                labels=['young', 'adult', 'middle', 'senior'])
        df['age_group_encoded'] = LabelEncoder().fit_transform(df['age_group'])
        
        # 5. Score composite
        df['composite_score'] = (
            (df['credit_score'] / 850) * 0.3 +
            (1 - df['debt_to_income']) * 0.25 +
            (df['income_to_loan_ratio'] / df['income_to_loan_ratio'].max()) * 0.2 +
            (df['employment_stability'] / df['employment_stability'].max()) * 0.15 +
            (df['financial_health'] / df['financial_health'].max()) * 0.1
        )
        
        self.feature_columns = (numerical_features + 
                               [f'{f}_encoded' for f in categorical_features] +
                               ['income_to_loan_ratio', 'credit_utilization', 
                                'employment_stability', 'financial_health',
                                'age_group_encoded', 'composite_score'])
        
        print(f"‚úÖ {len(self.feature_columns)} caract√©ristiques cr√©√©es")
        return df
    
    def train_model(self, df):
        """Entra√Ænement du mod√®le de scoring"""
        print("ü§ñ Entra√Ænement du mod√®le en cours...")
        
        # Pr√©paration des donn√©es
        X = df[self.feature_columns]
        y = df['default_risk']
        
        # Normalisation
        X_scaled = self.scaler.fit_transform(X)
        
        # Division train/test
        X_train, X_test, y_train, y_test = train_test_split(
            X_scaled, y, test_size=0.2, random_state=42, stratify=y
        )
        
        # Entra√Ænement
        self.model.fit(X_train, y_train)
        
        # √âvaluation
        train_score = self.model.score(X_train, y_train)
        test_score = self.model.score(X_test, y_test)
        
        print(f"üìà Score d'entra√Ænement: {train_score:.3f}")
        print(f"üìà Score de test: {test_score:.3f}")
        
        # Importance des caract√©ristiques
        feature_importance = pd.DataFrame({
            'feature': self.feature_columns,
            'importance': self.model.feature_importances_
        }).sort_values('importance', ascending=False)
        
        print("\nüèÜ Top 10 des caract√©ristiques les plus importantes:")
        print(feature_importance.head(10))
        
        return feature_importance
    
    def predict_risk(self, customer_data):
        """Pr√©diction du risque pour un nouveau client"""
        print("üéØ √âvaluation du risque client...")
        
        # Transformation des donn√©es
        customer_df = pd.DataFrame([customer_data])
        
        # Feature engineering pour le nouveau client
        customer_df['income_to_loan_ratio'] = customer_df['income'] / (customer_df['loan_amount'] + 1)
        customer_df['credit_utilization'] = customer_df['debt_to_income'] * customer_df['account_balance']
        customer_df['employment_stability'] = customer_df['employment_length'] / (customer_df['age'] - 18 + 1)
        customer_df['financial_health'] = (customer_df['credit_score'] / 850) * (1 - customer_df['debt_to_income'])
        
        # Cat√©gorisation de l'√¢ge
        customer_df['age_group'] = pd.cut(customer_df['age'], bins=[0, 25, 35, 50, 100], 
                                         labels=['young', 'adult', 'middle', 'senior'])
        customer_df['age_group_encoded'] = LabelEncoder().fit_transform(customer_df['age_group'])
        
        # Score composite
        customer_df['composite_score'] = (
            (customer_df['credit_score'] / 850) * 0.3 +
            (1 - customer_df['debt_to_income']) * 0.25 +
            (customer_df['income_to_loan_ratio'] / customer_df['income_to_loan_ratio'].max()) * 0.2 +
            (customer_df['employment_stability'] / customer_df['employment_stability'].max()) * 0.15 +
            (customer_df['financial_health'] / customer_df['financial_health'].max()) * 0.1
        )
        
        # Encodage des variables cat√©gorielles
        for feature in ['payment_history']:
            if feature in customer_data:
                customer_df[f'{feature}_encoded'] = self.label_encoders[feature].transform([customer_data[feature]])
        
        # Pr√©diction
        X = customer_df[self.feature_columns]
        X_scaled = self.scaler.transform(X)
        
        risk_probability = self.model.predict_proba(X_scaled)[0][1]
        risk_score = int(risk_probability * 1000)
        
        # Classification du risque
        if risk_score < 300:
            risk_category = "Faible"
            recommendation = "Approuv√©"
        elif risk_score < 600:
            risk_category = "Mod√©r√©"
            recommendation = "Approuv√© avec conditions"
        elif risk_score < 800:
            risk_category = "√âlev√©"
            recommendation = "Rejet√©"
        else:
            risk_category = "Tr√®s √©lev√©"
            recommendation = "Rejet√©"
        
        result = {
            'risk_score': risk_score,
            'risk_probability': risk_probability,
            'risk_category': risk_category,
            'recommendation': recommendation,
            'confidence': max(self.model.predict_proba(X_scaled)[0])
        }
        
        print(f"üìä Score de risque: {risk_score}/1000")
        print(f"üìä Cat√©gorie: {risk_category}")
        print(f"üìä Recommandation: {recommendation}")
        
        return result

from pyspark.sql import SparkSession
from pyspark.sql.functions import *
from pyspark.sql.types import *
from pyspark.ml import Pipeline
from pyspark.ml.feature import VectorAssembler, StandardScaler, StringIndexer
from pyspark.ml.classification import RandomForestClassifier
from pyspark.ml.evaluation import BinaryClassificationEvaluator
import findspark
findspark.init()

class SparkBankingRiskProcessor:
    """Processeur distribu√© pour l'√©valuation des risques bancaires"""
    
    def __init__(self):
        self.spark = SparkSession.builder \
            .appName("BankingRiskAssessment") \
            .config("spark.sql.adaptive.enabled", "true") \
            .config("spark.sql.adaptive.coalescePartitions.enabled", "true") \
            .getOrCreate()
        
        self.spark.conf.set("spark.sql.adaptive.skewJoin.enabled", "true")
        self.spark.conf.set("spark.serializer", "org.apache.spark.serializer.KryoSerializer")
        
        print("üöÄ Spark Session initialis√©e pour le traitement bancaire")
    
    def create_banking_schema(self):
        """D√©finition du sch√©ma pour les donn√©es bancaires"""
        schema = StructType([
            StructField("customer_id", StringType(), True),
            StructField("transaction_id", StringType(), True),
            StructField("transaction_date", TimestampType(), True),
            StructField("transaction_amount", DoubleType(), True),
            StructField("transaction_type", StringType(), True),
            StructField("merchant_category", StringType(), True),
            StructField("account_balance_before", DoubleType(), True),
            StructField("account_balance_after", DoubleType(), True),
            StructField("is_fraudulent", IntegerType(), True),
            StructField("risk_score", DoubleType(), True)
        ])
        return schema
    
    def load_banking_data(self, file_path):
        """Chargement des donn√©es bancaires avec Spark"""
        print("üìä Chargement des donn√©es bancaires avec Spark...")
        
        schema = self.create_banking_schema()
        
        # Chargement des donn√©es
        df = self.spark.read \
            .option("header", "true") \
            .option("inferSchema", "false") \
            .schema(schema) \
            .csv(file_path)
        
        # Cache pour optimiser les performances
        df.cache()
        
        print(f"‚úÖ {df.count()} transactions charg√©es")
        return df
    
    def feature_engineering_spark(self, df):
        """Feature engineering distribu√© avec Spark"""
        print("üîß Feature Engineering distribu√© en cours...")
        
        # 1. Nettoyage des donn√©es
        df_clean = df.filter(col("transaction_amount").isNotNull()) \
                    .filter(col("transaction_amount") > 0)
        
        # 2. Variables temporelles
        df_with_time = df_clean.withColumn("hour", hour("transaction_date")) \
                              .withColumn("day_of_week", dayofweek("transaction_date")) \
                              .withColumn("month", month("transaction_date"))
        
        # 3. Agr√©gations par client
        customer_stats = df_with_time.groupBy("customer_id") \
            .agg(
                count("*").alias("total_transactions"),
                sum("transaction_amount").alias("total_amount"),
                avg("transaction_amount").alias("avg_transaction"),
                max("transaction_amount").alias("max_transaction"),
                min("transaction_amount").alias("min_transaction"),
                stddev("transaction_amount").alias("transaction_std"),
                countDistinct("merchant_category").alias("unique_merchants"),
                countDistinct("transaction_type").alias("unique_types")
            )
        
        # 4. D√©tection d'anomalies temporelles
        time_anomalies = df_with_time.groupBy("customer_id", "hour") \
            .agg(count("*").alias("transactions_per_hour")) \
            .groupBy("customer_id") \
            .agg(stddev("transactions_per_hour").alias("hour_variability"))
        
        # 5. Patterns de d√©penses
        spending_patterns = df_with_time.groupBy("customer_id") \
            .pivot("merchant_category") \
            .agg(sum("transaction_amount").alias("amount")) \
            .fillna(0)
        
        # 6. Variables de risque
        risk_features = df_with_time.groupBy("customer_id") \
            .agg(
                sum(when(col("is_fraudulent") == 1, 1).otherwise(0)).alias("fraud_count"),
                avg("risk_score").alias("avg_risk_score"),
                max("risk_score").alias("max_risk_score")
            )
        
        # 7. Jointure de toutes les caract√©ristiques
        feature_df = customer_stats \
            .join(time_anomalies, "customer_id", "left") \
            .join(spending_patterns, "customer_id", "left") \
            .join(risk_features, "customer_id", "left") \
            .fillna(0)
        
        print(f"‚úÖ Feature engineering termin√©: {feature_df.count()} clients trait√©s")
        return feature_df
    
    def train_ml_model_spark(self, df):
        """Entra√Ænement du mod√®le ML distribu√©"""
        print("ü§ñ Entra√Ænement du mod√®le ML distribu√©...")
        
        # Pr√©paration des colonnes
        feature_columns = [col for col in df.columns if col != "customer_id" and col != "fraud_count"]
        
        # Vectorisation des caract√©ristiques
        assembler = VectorAssembler(
            inputCols=feature_columns,
            outputCol="features"
        )
        
        # Normalisation
        scaler = StandardScaler(
            inputCol="features",
            outputCol="scaled_features"
        )
        
        # Mod√®le Random Forest
        rf = RandomForestClassifier(
            featuresCol="scaled_features",
            labelCol="fraud_count",
            numTrees=100,
            maxDepth=10,
            seed=42
        )
        
        # Pipeline ML
        pipeline = Pipeline(stages=[assembler, scaler, rf])
        
        # Division train/test
        train_df, test_df = df.randomSplit([0.8, 0.2], seed=42)
        
        # Entra√Ænement
        model = pipeline.fit(train_df)
        
        # Pr√©dictions
        predictions = model.transform(test_df)
        
        # √âvaluation
        evaluator = BinaryClassificationEvaluator(
            labelCol="fraud_count",
            rawPredictionCol="rawPrediction"
        )
        
        auc = evaluator.evaluate(predictions)
        print(f"üìà AUC Score: {auc:.3f}")
        
        # Importance des caract√©ristiques
        rf_model = model.stages[-1]
        feature_importance = rf_model.featureImportances.toArray()
        
        importance_df = self.spark.createDataFrame([
            (feature, float(importance)) 
            for feature, importance in zip(feature_columns, feature_importance)
        ], ["feature", "importance"]) \
        .orderBy(desc("importance"))
        
        print("\nüèÜ Top 10 des caract√©ristiques les plus importantes:")
        importance_df.show(10)
        
        return model, importance_df
    
    def real_time_risk_assessment(self, df, model):
        """√âvaluation des risques en temps r√©el"""
        print("‚ö° √âvaluation des risques en temps r√©el...")
        
        # Traitement des nouvelles transactions
        new_transactions = df.filter(col("transaction_date") >= current_timestamp() - expr("INTERVAL 1 HOUR"))
        
        # Feature engineering pour les nouvelles donn√©es
        new_features = self.feature_engineering_spark(new_transactions)
        
        # Pr√©dictions
        predictions = model.transform(new_features)
        
        # Alertes de risque √©lev√©
        high_risk_alerts = predictions.filter(col("prediction") == 1.0) \
                                     .select("customer_id", "prediction", "probability")
        
        alert_count = high_risk_alerts.count()
        print(f"üö® {alert_count} alertes de risque √©lev√© d√©tect√©es")
        
        return high_risk_alerts
    
    def generate_risk_report(self, df, model):
        """G√©n√©ration de rapports de risque"""
        print("üìä G√©n√©ration des rapports de risque...")
        
        # Analyse globale des risques
        global_risk = df.groupBy("customer_id") \
            .agg(
                avg("risk_score").alias("avg_risk"),
                max("risk_score").alias("max_risk"),
                count("*").alias("transaction_count")
            )
        
        # Classification des clients par niveau de risque
        risk_classification = global_risk.withColumn("risk_category",
            when(col("avg_risk") < 300, "Faible")
            .when(col("avg_risk") < 600, "Mod√©r√©")
            .when(col("avg_risk") < 800, "√âlev√©")
            .otherwise("Tr√®s √©lev√©")
        )
        
        # Statistiques par cat√©gorie
        risk_stats = risk_classification.groupBy("risk_category") \
            .agg(
                count("*").alias("client_count"),
                avg("avg_risk").alias("avg_risk_score"),
                sum("transaction_count").alias("total_transactions")
            ) \
            .orderBy("avg_risk_score")
        
        print("üìà Statistiques des risques par cat√©gorie:")
        risk_stats.show()
        
        return risk_classification, risk_stats
    
    def stop_spark(self):
        """Arr√™t de la session Spark"""
        self.spark.stop()
        print("üõë Session Spark arr√™t√©e")

# Utilisation du processeur Spark
if __name__ == "__main__":

    def generate_daily_report(self):
        """G√©n√©ration de rapports quotidiens"""
        print("üìä G√©n√©ration du rapport quotidien...")
        
        # Requ√™te BigQuery pour les statistiques du jour
        query = """
        SELECT 
            DATE(transaction_date) as report_date,
            COUNT(*) as total_transactions,
            COUNT(DISTINCT customer_id) as unique_customers,
            SUM(CASE WHEN is_fraudulent = 1 THEN 1 ELSE 0 END) as fraudulent_transactions,
            AVG(risk_score) as avg_risk_score,
            SUM(transaction_amount) as total_volume
        FROM `{}.banking_risk_data.customer_transactions`
        WHERE DATE(transaction_date) = CURRENT_DATE()
        GROUP BY DATE(transaction_date)
        """.format(self.project_id)
        
        try:
            query_job = self.bq_client.query(query)
            results = query_job.result()
            
            report_data = []
            for row in results:
                report_data.append({
                    "report_date": row.report_date,
                    "total_transactions": row.total_transactions,
                    "unique_customers": row.unique_customers,
                    "fraudulent_transactions": row.fraudulent_transactions,
                    "avg_risk_score": row.avg_risk_score,
                    "total_volume": row.total_volume
                })
            
            # Sauvegarde du rapport
            report_df = pd.DataFrame(report_data)
            report_path = f"gs://banking-reports/daily_reports/{datetime.now().strftime('%Y%m%d')}_report.csv"
            
            report_df.to_csv(report_path, index=False)
            print(f"‚úÖ Rapport quotidien g√©n√©r√©: {report_path}")
            
            return report_data
            
        except Exception as e:
            print(f"‚ùå Erreur lors de la g√©n√©ration du rapport: {e}")
            return None
